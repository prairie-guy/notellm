import matplotlib.pyplot as plt
from IPython.utils.capture import capture_output

# Convert to pandas for seaborn (seaborn works best with pandas)

with capture_output() as _claude_captured_output:

from Bio import SeqIO

# Read the E. coli FASTA file
fasta_file = "ecoli_k12_mg1655.fna"

# Parse the FASTA file and calculate GC content
for record in SeqIO.parse(fasta_file, "fasta"):
    sequence = str(record.seq)
    
    # Count bases
    total_length = len(sequence)
    g_count = sequence.count('G')
    c_count = sequence.count('C')
    a_count = sequence.count('A')
    t_count = sequence.count('T')
    
    # Calculate GC content
    gc_count = g_count + c_count
    gc_percentage = (gc_count / total_length) * 100
    
    # Display results
    results = {
        'Sequence ID': record.id,
        'Description': record.description,
        'Total Length (bp)': f"{total_length:,}",
        'A count': f"{a_count:,}",
        'T count': f"{t_count:,}",
        'G count': f"{g_count:,}",
        'C count': f"{c_count:,}",
        'GC count': f"{gc_count:,}",
        'GC Content (%)': f"{gc_percentage:.2f}%"
    }
    
    print("E. coli K-12 MG1655 Genome Statistics")
    print("=" * 60)
    for key, value in results.items():
        print(f"{key:20s}: {value}")
    
    # Return the GC percentage
    gc_percentage
